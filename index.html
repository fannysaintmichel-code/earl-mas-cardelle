function importPAC(input){ let file = input.files[0]; let reader = new FileReader(); reader.onload = function(e){ let geo = JSON.parse(e.target.result); geo.features.forEach(f=>{ let coords = f.geometry.coordinates[0].map(c=>[c[1],c[0]]); let surf = surface(coords).toFixed(2); parcelles.push({ id:Date.now()+Math.random(), nom:"PAC", culture:"", surface:surf, points:coords, interventions:[] }); L.polygon(coords,{color:'blue'}).addTo(map); }); localStorage.setItem("parcelles",JSON.stringify(parcelles)); update(); }; reader.readAsText(file); } function sauvegardeCloud(){ let data = JSON.stringify(parcelles); navigator.clipboard.writeText(data); alert("Sauvegarde copiée ! Colle-la ailleurs (WhatsApp, mail…) 👍"); }
